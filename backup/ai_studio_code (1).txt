import React from 'react';
import { motion } from 'framer-motion';

interface GlassCardProps {
  children: React.ReactNode;
  className?: string;
  glowColor?: 'cyan' | 'gold' | 'red' | 'none';
}

export const GlassCard: React.FC<GlassCardProps> = ({ children, className = '', glowColor = 'none' }) => {
  const glowMap = {
    cyan: 'hover:shadow-glow-cyan',
    gold: 'hover:shadow-glow-gold',
    red: 'hover:shadow-glow-red',
    none: 'hover:shadow-xl'
  };

  return (
    <motion.div
      whileHover={{ scale: 1.02 }}
      transition={{ type: "spring", stiffness: 400, damping: 25 }}
      className={`relative rounded-3xl bg-cm-panel backdrop-blur-apple border border-white/10 ${glowMap} transition-shadow duration-300 overflow-hidden ${className}`}
    >
      {/* Subtle top highlight to simulate glass edge */}
      <div className="absolute top-0 left-0 w-full h- bg-gradient-to-r from-transparent via-white/20 to-transparent" />
      <div className="p-6 relative z-10">
        {children}
      </div>
    </motion.div>
  );
};
